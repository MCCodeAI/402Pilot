\section{Notation}
\label{appendix:notation}

This appendix collects every symbol used in the paper on a single page so that
reviewers and re-implementers can cross-reference Sections~\ref{sec:problem},
\ref{sec:method}, and~\ref{sec:eval} against a single legend. Locked numerical
values match the YAML configuration shipped with the released artifact.

\subsection{Per-round observations}
\label{app:notation-obs}

\begin{tabularx}{\columnwidth}{@{}l>{\raggedright\arraybackslash}X>{\raggedright\arraybackslash}X@{}}
\toprule
Symbol & Meaning & Domain / value \\
\midrule
$q_t$       & Quality score for chosen arm & $[0,1]$ (evaluator-grounded) \\
$\bar c_a$  & Nominal listed cost for arm $a$ & known before the run; prior mean for C-posterior \\
$p_{a,t}$   & Effective charge if arm $a$ is called & latent before selection; may drift \\
$\mathcal{Q}_t$ & Quoted candidate set & provider-price pairs $\{(a,p_{a,t})\}_{a=1}^K$ from upstream discovery/pricing \\
$\mathcal{A}_t$ & Affordable provider set & $\{a:p_{a,t}\le B_t\}$, exposed to policy \\
$c_t$       & Raw cost charged this round (USDC) & selected charge $p_{a_t,t}$ \\
$\ctilde$   & Normalized cost & $\mathrm{clip}(c_t/c_{\max},0,1)$, $c_{\max} = \$0.01$ \\
$f_t$       & Failure flag (timeout / parse-failure) & $\{0,1\}$ \\
$B$         & Total wallet budget & $B = \$50$ \\
$B_t$       & Remaining wallet budget at start of round $t$ & $B_0 = B$, $B_{t+1} = B_t - c_t$ \\
\bottomrule
\end{tabularx}

Bandit feedback: only $(q_t,c_t,f_t)$ for $a = a_t$ is observed; outcomes and
costs for arms not chosen are unseen.

\subsection{Reward and utility}
\label{app:notation-reward}

\begin{tabularx}{\columnwidth}{@{}l>{\raggedright\arraybackslash}X>{\raggedright\arraybackslash}X@{}}
\toprule
Symbol & Definition & Notes \\
\midrule
$\nu$       & Failure penalty constant & $\nu = 0.5$ (locked) \\
$\ut$       & Utility, $\ut = q_t - \nu f_t$ & range $[-0.5,1]$; updates Q-posterior \\
$\alpha$    & Sigmoid sharpness for budget pressure & $\alpha = 2.0$ (locked) \\
$\lambda_0$ & Baseline budget-pressure multiplier & $\lambda_0 = 1$ (locked) \\
$\bdev_t$   & Normalized burn deviation & signed deviation from target burn rate \\
$\lambda_t$ & Raw budget-pressure multiplier & $\lambda_t = \lambda_0\exp(\alpha\,\bdev_t) > 0$ for $t>0$ \\
$\lamn$     & Bounded sigmoid weight & $\lamn = \lambda_t/(1+\lambda_t) \in (0,1)$ \\
$\rt$       & Payment-aware reward & $\rt = (1-\lamn)\,\ut - \lamn\,\ctilde \in [-1,1]$ \\
\bottomrule
\end{tabularx}

\paragraph{Critical separation.} $\ut$ updates the policy's belief about a
provider; $\rt$ ranks arms at decision time. The decoupling keeps beliefs
stable when $\lambda_t$ jumps (\S\ref{sec:layer}).

\subsection{Posteriors}
\label{app:notation-posteriors}

For each arm $a \in \{1,\ldots,K\}$ and bucket $k \in \{1,\ldots,K_b\}$ two
independent Normal--Normal Gaussian posteriors are maintained.

\paragraph{Q-posterior (quality).}
Prior $\mathcal{N}(\mu_{0,q},\sigma^2_{0,q})$ with $\mu_{0,q} = 0.5$,
$\sigma^2_{0,q} = 1.0$; likelihood variance $\sigma^2_q = 0.09$. Sufficient
statistics $(n^{a,k}_t, S^{a,k}_t)$ are the discounted effective sample count
and weighted sum of utilities. Closed forms:
\[
\begin{aligned}
\Sigmaq{a}{k}
  &= \!\left(\tfrac{1}{\sigma^2_{0,q}}
  + \tfrac{n^{a,k}_t}{\sigma^2_q}\right)^{\!-1},\\
\thetaq{a}{k}
  &= \Sigmaq{a}{k}\!\left(\tfrac{\mu_{0,q}}{\sigma^2_{0,q}}
  + \tfrac{S^{a,k}_t}{\sigma^2_q}\right).
\end{aligned}
\]
At decision time, $\hatu{a}{k}\sim\mathcal{N}(\thetaq{a}{k},\Sigmaq{a}{k})$.

\paragraph{C-posterior (cost).}
Prior $\mathcal{N}(\bar c_a, \sigma^2_{0,c})$ where $\bar c_a$ is the spec base
price for arm $a$; $\sigma^2_{0,c} = 10^{-4}$, $\sigma^2_c = 10^{-6}$ (tight
because realized costs are nearly deterministic given provider and scenario).
Sufficient statistics $(n^{a,k}_t, {S_c}^{a,k}_t)$ over realized costs;
posterior mean and variance follow the same Normal--Normal closed form with
$\bar c_a$ in place of $\mu_{0,q}$, etc. At decision time,
$\hatc{a}{k}\sim\mathcal{N}(\thetac{a}{k},\Sigmac{a}{k})$, normalized as
$\tilde{\hat c}^{a,k} = \mathrm{clip}(\hatc{a}{k}/c_{\max},0,1)$.

\paragraph{Discounted sufficient statistics update.}
At the start of every round, every $(a,k)$ cell discounts:
\[
\begin{aligned}
n^{a,k}_t &\leftarrow \gamma\, n^{a,k}_{t-1},\\
S^{a,k}_t &\leftarrow \gamma\, S^{a,k}_{t-1},
\end{aligned}
\]
with the analogous sweep on the cost statistics. Then the chosen arm's bucket
increments: $n^{a_t,k_t}_t\mathrel{+}=1$, $S^{a_t,k_t}_t\mathrel{+}=\ut$,
${S_c}^{a_t,k_t}_t\mathrel{+}=c_t$. Discounting applies regardless of pulls
(non-stationarity is wall-clock, not pull-conditional).

\subsection{Decision rule}
\label{app:notation-decision}

For each affordable arm $a \in \mathcal{A}_t$:
\[
\begin{aligned}
\hatu{a}{k_t}
&\sim \mathcal{N}(\thetaq{a}{k_t},\Sigmaq{a}{k_t}),\\
\hatc{a}{k_t}
&\sim \mathcal{N}(\thetac{a}{k_t},\Sigmac{a}{k_t}),
\end{aligned}
\]
\[
\begin{aligned}
\tilde{\hat c}^{a,k_t}
&= \mathrm{clip}(\hatc{a}{k_t}/c_{\max},0,1),\\
s_a
&= (1-\lamn)\,\hatu{a}{k_t}
  - \lamn\,\tilde{\hat c}^{a,k_t}.
\end{aligned}
\]
The chosen arm is $a_t = \arg\max_{a\in\mathcal{A}_t}s_a$. Under $-TS$
replace $\sim$ with $=$ (use posterior means). Under $-P$ force $\lamn = 0$
in $s_a$.

\subsection{Metrics}
\label{app:notation-metrics}

\begin{tabularx}{\columnwidth}{@{}l>{\raggedright\arraybackslash}X@{}}
\toprule
Symbol & Definition / notes \\
\midrule
$\bar q_T$       & $(1/T)\sum_{t<\tau} q_t$; full-horizon mean quality, with unserved rounds after wallet exhaustion counted as $q_t = 0$ \\
$\mathrm{budget\ use}$ & $100\sum_{t<\tau} c_t/B$; percentage of the initial wallet spent \\
$\mathrm{ROI}$   & $\sum_{t<\tau} q_t / \sum_{t<\tau} c_t$; quality yield per dollar \\
$R^\pi_T$        & $\sum_{t<\tau} r_t$; cumulative payment-aware reward optimized by PA-DCT \\
$\paReg_T$       & $R^{\mathrm{Oracle}}_T - R^\pi_T$; empirical PA-regret to the True Oracle, Eq.~\eqref{eq:pa-regret} \\
$\paReg_T/T$     & $(R^{\mathrm{Oracle}}_T - R^\pi_T)/T$; reported as PA-gap/$T$ \\
$\AdaptT$        & trailing-200 ROI shock-response time; S2 reaches $\geq 95\%$ of pre-outage ROI, S3 reaches $\geq 110\%$ of pre-promotion ROI \\
\bottomrule
\end{tabularx}

\subsection{Ablation labels}
\label{app:notation-ablations}

\begin{tabularx}{\columnwidth}{@{}l>{\raggedright\arraybackslash}X>{\raggedright\arraybackslash}X@{}}
\toprule
Label & Means & Effect on PA-DCT \\
\midrule
$-P$            & no payment-aware ranking  & force $\lamn = 0$ in $s_a$ \\
$-D$            & no discount               & $\gamma = \gamma_c = 1$ \\
$-C$            & no contextual bucketing   & $K_b = 1$ (single global bucket) \\
$-TS$           & no Thompson sampling      & rank arms by posterior means $\thetaq{a}{k}, \thetac{a}{k}$ \\
$-C_{\mathrm{post}}$ & no learned cost posterior & pin $\hatc{a}{k} = \bar c_a$; skip C-posterior discount and update \\
\bottomrule
\end{tabularx}

The ``full'' configuration enables all five; each ablation flips exactly one
flag and leaves the others at their default values.
